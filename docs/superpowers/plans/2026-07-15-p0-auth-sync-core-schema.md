# P0: Auth Sync + Core Schema + FE Backend Utility — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Set up the foundation: database with core schema, auth sync between Supertokens and core.users, FE backendFetch utility, and a basic getProfile endpoint to prove the full vertical slice works.

**Architecture:** PostgreSQL schema `core` with `users` + `user_stats` tables, sqlc type-safe query layer, Kratos HTTP/gRPC server exposing ProfileService, Next.js Server Action calling Go backend via backendFetch with SuperTokens session authentication.

**Tech Stack:** Go 1.26, Kratos v3, sqlc + pgx/v5, goose (migrations), PostgreSQL 18 (CNPG on K3s), Next.js 16 + React 19, Supertokens (existing).

```go
// app/core/internal/biz/profile.go
package biz

import (
    "context"
    "fmt"
    "regexp"
    "strings"

    "github.com/puchidemy/puchi-backend/app/core/internal/data/sqlc/gen"
)

// ProfileInput for update
type UpdateProfileInput struct {
    FirstName string
    LastName  string
    Username  string
    Bio       string
}

// UserRepoInterface — defined in biz layer (dependency inversion)
type UserRepoInterface interface {
    CreateUser(ctx context.Context, arg gen.CreateUserParams) (*gen.User, error)
    GetUser(ctx context.Context, id string) (*gen.User, error)
    GetUserByEmail(ctx context.Context, email string) (*gen.User, error)
    UpdateUser(ctx context.Context, arg gen.UpdateUserParams) (*gen.User, error)
    UsernameExists(ctx context.Context, username string) (bool, error)
}

// Domain errors
var (
    ErrUserNotFound  = fmt.Errorf("user not found")
    ErrUsernameTaken = fmt.Errorf("username already taken")
)

type ProfileUsecase struct {
    repo UserRepoInterface
}

func NewProfileUsecase(repo UserRepoInterface) *ProfileUsecase {
    return &ProfileUsecase{repo: repo}
}

func (uc *ProfileUsecase) GetProfile(ctx context.Context, userID string) (*gen.User, error) {
    user, err := uc.repo.GetUser(ctx, userID)
    if err != nil {
        return nil, fmt.Errorf("get user: %w", err)
    }
    return user, nil
}

func (uc *ProfileUsecase) UpdateProfile(ctx context.Context, userID string, input UpdateProfileInput) (*gen.User, error) {
    // Check username uniqueness
    exists, err := uc.repo.UsernameExists(ctx, input.Username)
    if err != nil {
        return nil, fmt.Errorf("check username: %w", err)
    }
    // Allow if username hasn't changed
    currentUser, _ := uc.repo.GetUser(ctx, userID)
    if exists && currentUser != nil && currentUser.Username != input.Username {
        return nil, ErrUsernameTaken
    }

    user, err := uc.repo.UpdateUser(ctx, gen.UpdateUserParams{
        ID:        userID,
        FirstName: input.FirstName,
        LastName:  input.LastName,
        Username:  input.Username,
        Bio:       stringPtr(input.Bio),
        AvatarKey: &currentUser.AvatarKey, // keep current
    })
    if err != nil {
        return nil, fmt.Errorf("update user: %w", err)
    }
    return user, nil
}

func (uc *ProfileUsecase) CreateUser(ctx context.Context, userID, username, email, firstName, lastName string) (*gen.User, error) {
    // Auto-generate username if empty
    if username == "" {
        username = generateUsername(email)
    }

    user, err := uc.repo.CreateUser(ctx, gen.CreateUserParams{
        ID:        userID,
        Username:  username,
        Email:     email,
        FirstName: firstName,
        LastName:  lastName,
    })
    if err != nil {
        return nil, fmt.Errorf("create user: %w", err)
    }
    return user, nil
}

func generateUsername(email string) string {
    localPart := strings.Split(email, "@")[0]
    localPart = regexp.MustCompile(`[^a-z0-9]`).ReplaceAllString(strings.ToLower(localPart), "")
    if len(localPart) < 3 {
        localPart = "puchi_user"
    }
    return localPart
}

func stringPtr(s string) *string {
    if s == "" {
        return nil
    }
    return &s
}
```

```go
// app/core/internal/service/profile.go
package service

import (
    "context"

    pb "github.com/puchidemy/puchi-backend/app/core/api/profile/v1"
    "github.com/puchidemy/puchi-backend/app/core/internal/auth"
    "github.com/puchidemy/puchi-backend/app/core/internal/biz"

    "google.golang.org/protobuf/types/known/timestamppb"
    "github.com/go-kratos/kratos/v3/errors"
)

type ProfileService struct {
    pb.UnimplementedProfileServiceServer
    uc *biz.ProfileUsecase
}

func NewProfileService(uc *biz.ProfileUsecase) *ProfileService {
    return &ProfileService{uc: uc}
}

func (s *ProfileService) GetProfile(ctx context.Context, _ *emptypb.Empty) (*pb.User, error) {
    userID, ok := auth.UserIDFromContext(ctx)
    if !ok {
        return nil, errors.Unauthorized("UNAUTHORIZED", "not logged in")
    }

    user, err := s.uc.GetProfile(ctx, userID)
    if err != nil {
        return nil, errors.NotFound("USER_NOT_FOUND", "user not found")
    }

    return userToProto(user), nil
}

func (s *ProfileService) UpdateProfile(ctx context.Context, req *pb.UpdateProfileRequest) (*pb.User, error) {
    userID, ok := auth.UserIDFromContext(ctx)
    if !ok {
        return nil, errors.Unauthorized("UNAUTHORIZED", "not logged in")
    }

    user, err := s.uc.UpdateProfile(ctx, userID, biz.UpdateProfileInput{
        FirstName: req.FirstName,
        LastName:  req.LastName,
        Username:  req.Username,
        Bio:       req.Bio,
    })
    if err == biz.ErrUsernameTaken {
        return nil, errors.AlreadyExists("USERNAME_TAKEN", err.Error())
    }
    if err != nil {
        return nil, errors.InternalServer("INTERNAL_ERROR", err.Error())
    }

    return userToProto(user), nil
}

func userToProto(u *gen.User) *pb.User {
    return &pb.User{
        Id:        u.ID,
        Username:  u.Username,
        FirstName: u.FirstName,
        LastName:  u.LastName,
        Email:     u.Email,
        Bio:       zeroPtrToString(u.Bio),
        CreatedAt: timestamppb.New(u.CreatedAt),
        UpdatedAt: timestamppb.New(u.UpdatedAt),
    }
}

func zeroPtrToString(s *string) string {
    if s == nil {
        return ""
    }
    return *s
}
```

- [ ] **Step 1: Create sync.go**

```go
// app/core/internal/auth/sync.go
package auth

import (
    "context"
    "fmt"

    "github.com/supertokens/supertokens-golang/recipe/session"
    "github.com/supertokens/supertokens-golang/supertokens"
    "github.com/puchidemy/puchi-backend/app/core/internal/data/sqlc/gen"
)

type UserSyncer struct {
    repo UserRepoInterface
}

type UserRepoInterface interface {
    CreateUser(ctx context.Context, arg gen.CreateUserParams) (*gen.User, error)
    GetUser(ctx context.Context, id string) (*gen.User, error)
    UsernameExists(ctx context.Context, username string) (bool, error)
}

func NewUserSyncer(repo UserRepoInterface) *UserSyncer {
    return &UserSyncer{repo: repo}
}

// EnsureUserExists checks if user exists in DB, creates if not (lazy creation)
func (s *UserSyncer) EnsureUserExists(ctx context.Context, userID string) error {
    _, err := s.repo.GetUser(ctx, userID)
    if err == nil {
        return nil // already exists
    }

    // User not found — lazy create from Supertokens
    info, err := supertokens.GetUser(userID)
    if err != nil {
        return fmt.Errorf("get user from supertokens: %w", err)
    }

    email := info.Email
    if email == "" {
        email = fmt.Sprintf("%s@unknown.puchi", userID[:8])
    }

    username := generateUsername(email)

    // Check uniqueness with counter
    base := username
    for i := 1; ; i++ {
        exists, _ := s.repo.UsernameExists(ctx, username)
        if !exists {
            break
        }
        username = fmt.Sprintf("%s%d", base, i)
    }

    _, err = s.repo.CreateUser(ctx, gen.CreateUserParams{
        ID:        userID,
        Username:  username,
        Email:     email,
        FirstName: "",
        LastName:  "",
    })
    return err
}

func generateUsername(email string) string {
    localPart := email[:strings.IndexByte(email, '@')]
    localPart = regexp.MustCompile(`[^a-z0-9]`).ReplaceAllString(strings.ToLower(localPart), "")
    if len(localPart) < 3 {
        localPart = "puchi_user"
    }
    return localPart
}
```

```typescript
// src/lib/backend.ts
"server-only";

import { getAppDirSession } from "supertokens-node/nextjs";

const BACKEND_URL = process.env.API_INTERNAL_URL || "http://localhost:8000";

export class BackendError extends Error {
  constructor(
    public status: number,
    public code: string,
    public path: string,
  ) {
    super(code);
    this.name = "BackendError";
  }
}

export async function backendFetch<T>(
  path: string,
  options: RequestInit & { timeout?: number } = {},
): Promise<T> {
  const session = await getAppDirSession().catch(() => null);
  const timeout = options.timeout ?? 15000;

  const controller = new AbortController();
  const timer = setTimeout(() => controller.abort(), timeout);

  try {
    const res = await fetch(`${BACKEND_URL}${path}`, {
      ...options,
      headers: {
        "Content-Type": "application/json",
        ...(session ? { Authorization: `Bearer ${session.getAccessToken()}` } : {}),
        ...options.headers,
      },
      signal: controller.signal,
    });

    if (!res.ok) {
      const body = await res.json().catch(() => ({}));
      throw new BackendError(res.status, body.code || "INTERNAL_ERROR", path);
    }

    return res.status === 204 ? undefined : res.json();
  } catch (err) {
    if (err instanceof BackendError) throw err;
    if ((err as Error).name === "AbortError") {
      throw new BackendError(408, "NETWORK_TIMEOUT", path);
    }
    throw new BackendError(0, "NETWORK_ERROR", path);
  } finally {
    clearTimeout(timer);
  }
}
```

```typescript
};

export function getErrorI18nKey(err: unknown): string {
  if (err instanceof BackendError) {
    return ERROR_MAP[err.code] || "errors.server.internalError";
  }
  return "errors.server.internalError";
}
```

```typescript
// src/actions/profile/get-profile.ts
"use server";

import { backendFetch, BackendError } from "@/lib/backend";
import { getErrorI18nKey } from "@/lib/errors";
import type { ActionResult } from "@/types/api";

export interface UserProfile {
  id: string;
  username: string;
  firstName: string;
  lastName: string;
  email: string;
  avatarUrl: string;
  bio: string;
  createdAt: string;
  updatedAt: string;
}

export async function getProfile(): Promise<ActionResult<UserProfile>> {
  try {
    const data = await backendFetch<UserProfile>("/v1/profile");
    return { success: true, data };
  } catch (err) {
    return { success: false, error: getErrorI18nKey(err) };
  }
}
```

- [ ] **Step 2: Create update-profile.ts**

```typescript
// src/actions/profile/update-profile.ts
"use server";

import { revalidatePath } from "next/cache";
import { backendFetch } from "@/lib/backend";
import { getErrorI18nKey } from "@/lib/errors";
import type { ActionResult } from "@/types/api";
import type { UserProfile } from "./get-profile";

export async function updateProfile(
  _prevState: ActionResult<UserProfile> | null,
  formData: FormData,
): Promise<ActionResult<UserProfile>> {
  const body = {
    firstName: String(formData.get("firstName") || ""),
    lastName: String(formData.get("lastName") || ""),
    username: String(formData.get("username") || ""),
    bio: String(formData.get("bio") || ""),
  };

  try {
    const data = await backendFetch<UserProfile>("/v1/profile", {
      method: "PUT",
      body: JSON.stringify(body),
    });
    revalidatePath("/[locale]/(protected)/(nav)/profile", "page");
    return { success: true, data };
  } catch (err) {
    return { success: false, error: getErrorI18nKey(err) };
  }
}
```
